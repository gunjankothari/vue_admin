const getGroup = (items, levels, addAtLevel) => {
    if (levels.length > 0 && addAtLevel > 0) {
        const child = items[levels[0]].child;
        levels.shift();
        addAtLevel--;
        return getGroup(child, levels, addAtLevel);
    } else {
        return items;
    }
}
